Code example

Here is my solution the kata Fun with trees: max sum from Codewars's site


function maxSum(root) {
  if (root === null)
    return 0;
  
  let l = 0, r = 0;
  if (!root.left && !root.right)
    return root.value;
                        
  if (root.left)
    l = root.value + maxSum(root.left);
  if (root.right)
    r = root.value + maxSum(root.right);
                        
  if (l > r)
    return l
  return r; 
}
